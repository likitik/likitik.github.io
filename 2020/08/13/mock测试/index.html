常用mock测试

public class MockClass {
    public String mockTestFunc(int i) {
        publicFunc("test");
        return privateFunc(i);
    }

    private String privateFunc(int i) {
        return "private func " + i;
    }
    public void publicFunc(String s){
        System.out.println("public func" + s);
    }


    protected void protectedFunc(){
        System.out.println("visible protected func");
    }
}

测试powermock私有方法

通过反射，测试私有方法

<dependency>
  <groupId>org.powermock</groupId>
  <artifactId>powermock-module-junit4</artifactId>
  <version>2.0.7</version>
  <scope>test</scope>
</dependency>
<dependency>
  <groupId>org.powermock</groupId>
  <artifactId>powermock-api-mockito2</artifactId>
  <version>2.0.2</version>
  <scope>test</scope>
</dependency>
// mock私有方法必须写
@RunWith(PowerMockRunner.class)
@PrepareForTest({MockClass.class})
public class MockPrivateClassTest
{

    /**
     * mock共有方法
     */
    @Test
    public void testMockPublic(){
        MockClass mockClass = Mockito.spy(new MockClass());
        Mockito.doNothing().when(mockClass).publicFunc(Mockito.anyString());
        System.out.println(mockClass.mockTestFunc(1));
    }
    /**
     * mock私有方法
     * @throws Exception
     */
    @Test
    public void testMockPrivate() throws Exception {
        MockClass mockClass = PowerMockito.spy(new MockClass());
        PowerMockito.when(mockClass, "privateFunc",Mockito.anyInt()).thenReturn("mock private test");
        String realResult = mockClass.mockTestFunc(1);

        Assert.assertEquals("mock private test", realResult);
    }

    /**
     * 反射私有方法
     * @throws NoSuchMethodException
     * @throws InvocationTargetException
     * @throws IllegalAccessException
     */
    @Test
    public void testMethod() throws NoSuchMethodException, InvocationTargetException, IllegalAccessException {
        Method method = MockClass.class.getDeclaredMethod("privateFunc",int.class);
        method.setAccessible(true);
        MockClass mockClass = new MockClass();
        System.out.println(method.invoke(mockClass, 2));

    }

    @Test
    public void testProtected(){
        MockClass mockClass = new MockClass();
        mockClass.protectedFunc();
    }
}
